JavaScript Classes

old way


function Pet(){
  this.animal = '';
  this.name = '';
  this.setAnimal = function(newAnimal){
    this.animal = newAnimal;
  }
  this.setName = function(newName){
    this.name = newName;
  }
}
let myDog = new Pet();
myDog.setAnimal('Dog');
myDog.setName('Socks');
console.log(myDog);

Class Way


class Pet {
  constructor(animal, name){
    this.animal = animal;
    this.name = name;
  }

  setAnimal(newAnimal){
    this.animal = newAnimal;
  }

  setName(newName){
    this.name = newName;
  }
}
let myDog = new Pet();
myDog.setAnimal('Dog');
myDog.setName('Socks');
console.log(myDog);

let myCat = new Pet('Cat', 'Henry');
console.log(myCat);
