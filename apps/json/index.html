JSON

JSON.parse()

take note the digits are not in quotes


// simple JSON string 
let mary = '{ "name":"Mary", "age":"45", "eyes":"blue" }';
// parse the string to an object
let maryObject = JSON.parse(mary);
console.log(maryObject); // app.js:5
let output = "";
for(param in maryObject){
  output += `${param} = ${maryObject[param]} \n`
}
console.log(output); // app.js:10

JSON.stringify()

used to flatten the object so it can be transmitted


// creating new dan object
let dan = new Object()
dan.name = 'Dan';
dan.age = '46';
dan.eyes = 'brown';
console.log(dan); // app.js:17
// flattening the dan object
console.log(JSON.stringify(dan)); // app.js:18

// fetch api
function getData(){
  fetch('https://gist.githubusercontent.com/natebass/b0a548425a73bdf8ea5c618149fe1fce/raw/f4231cd5961f026264bb6bb3a6c41671b044f1f4/quotes.json')
  .then((res) => res.json())
  .then((data) => {
    console.log(data);
    let output = '
quotes
';
    data.forEach(function(quote, index){
      output += `
      
    
        
  • Index: ${index}
    • 
        
  • Qoute: ${quote.quote}
    • 
        
  • Author: ${quote.author}
    • 
      

      `;
    });
    document.getElementById('quote-div').innerHTML = output;
  })
  .catch((err) => {
    console.log(err);
  });
}
getData();
