// setInterval
let now = new Date();
let hours = now.getHours();
let minutes = now.getMinutes();
let seconds = now.getSeconds();

function setClock(){
  if(hours > 12){
    hours = hours - 12;
  }
  if(minutes === 60){
    hours ++;
    minutes = 0;
  }
  if(seconds === 60){
    minutes ++;
    seconds = 0;
  }

    //display time
  document.getElementById("clock").innerHTML = `${hours}:${("0" + minutes).slice(-2)}:${("0" + seconds).slice(-2)}`;

}
function incSeconds(){
  seconds ++;
  setClock();
}
setClock();
window.setInterval(incSeconds, 1000);
